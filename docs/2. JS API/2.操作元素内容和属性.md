### 操作元素内容

```js
.innerText	// 
```

```js
.innerHTML
```



```html
<div class="box">我是大王</div>
<script>
	const box = document.querySelector(".box")
    box.innerText = "我才是大王"	// 页面内容变成 我才是大王
</script>
```



```html
<div class="box">我是大王</div>
<script>
	const box = document.querySelector(".box")
    box.innerText = "<strong>我才是大王</strong>"	// 页面内容变成 <strong>我才是大王</strong>
    box.innerHTML = "<strong>我才是大王</strong>"	// 页面内容变成 我才是大王 加粗
</script>
```
